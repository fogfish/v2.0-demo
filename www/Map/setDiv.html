map.setDiv()

    Click the button, detach the map. Click the button again, attach the map. 

    var div = document.getElementById("map_canvas");
var map = plugin.google.maps.Map.getMap(div);
map.one(plugin.google.maps.event.MAP_READY, function() {

  var button = div.getElementsByTagName('button')[0];
  button.addEventListener('click', function() {
    if (map.getDiv()) {
      map.setDiv();
    } else {
      map.setDiv(div);
    }
  });

});

    Init map at offscreen element, Click the button, detach the map. 

    var div = document.getElementById("map_canvas_offscreen");
var offscreen = document.createElement('div');
offscreen.style.width = div.style.width;
offscreen.style.height = div.style.height;

var map = plugin.google.maps.Map.getMap(offscreen);
map.one(plugin.google.maps.event.MAP_READY, function() {

  var buttonA0 = div.getElementsByTagName('button')[0];

    map.setDiv(div)
